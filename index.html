Supabase SQL Setup

Führe dieses SQL in deinem Supabase-Projekt aus: 
CREATE TABLE IF NOT EXISTS kostenvoranschlaege (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  kv_nr text,
  kunde text,
  adresse text,
  telefon text,
  email text,
  status text DEFAULT 'entwurf',
  fahrzeug text,
  kennzeichen text,
  fin text,
  positionen jsonb,
  mat_gesamt numeric,
  lohn_gesamt numeric,
  netto numeric,
  brutto numeric,
  db_betrag numeric,
  db_prozent numeric,
  anmerkungen text,
  ki_analyse text,
  created_at timestamptz DEFAULT now()
);
ALTER TABLE kostenvoranschlaege ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public rw" ON kostenvoranschlaege FOR ALL USING (true) WITH CHECK (true);
